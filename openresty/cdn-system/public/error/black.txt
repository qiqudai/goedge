    <script>
        const translations = {
            en: {
                title: "Access Blocked",
                blacklistNotice: "Your IP address has been blacklisted",
                detailMessage: "Your access request has been denied due to suspicious activity."
            },
            zh: {
                title: "访问已被阻止",
                blacklistNotice: "您的 IP 地址已被列入黑名单",
                detailMessage: "由于检测到可疑活动，您的访问请求已被系统拒绝。"
            }
        };

        // 获取浏览器语言
        function getBrowserLanguage() {
            const language = navigator.language || navigator.userLanguage;
            // 简单处理中文和英文，其他语言默认使用英文
            return language.toLowerCase().startsWith("zh") ? "zh" : "en";
        }

        // 更新页面内容
        function updateContent(lang) {
            const content = translations[lang];
            document.getElementById("title").textContent = content.title;
            document.getElementById("blacklist-notice").textContent = content.blacklistNotice;
            document.getElementById("detail-message").textContent = content.detailMessage;
            document.documentElement.lang = lang;
            document.title = content.title;
        }

        // 初始化页面
        updateContent(getBrowserLanguage());
    </script>
