    <script>
        const translations = {
            en: {
                title: "Security Verification",
                placeholder: "Enter result",
                submit: "Submit",
                newProblem: "New Problem",
                altText: "Math Problem"
            },
            zh: {
                title: "安全验证",
                placeholder: "输入计算结果",
                submit: "提交答案",
                newProblem: "换一题",
                altText: "算术题"
            }
        };

        function detectLanguage() {
            const userLang = navigator.language || navigator.userLanguage;
            return userLang.startsWith("zh") ? "zh" : "en";
        }

        const currentLang = detectLanguage();

        function updateUI() {
            const texts = translations[currentLang];
            document.title = texts.title;
            document.querySelector("h2").textContent = texts.title;
            document.getElementById("answer").placeholder = texts.placeholder;
            document.querySelector(".submit-btn").textContent = texts.submit;
            document.querySelector(".refresh-icon").title = texts.newProblem;
            document.getElementById("captcha").alt = texts.altText;
            document.documentElement.lang = currentLang;
        }

        function refreshCaptcha() {
            const img = document.getElementById("captcha");
            const timestamp = new Date().getTime();
            img.src = `/_guard/captcha.png?t=${timestamp}&lang=${currentLang}`;
            document.getElementById("answer").focus();
        }

        window.onload = function() {
            updateUI();
            refreshCaptcha();
            document.getElementById("answer").focus();
        };
    </script>
